function createCdToRegionMap(regions: Region[]): Record<string, Region> {
  // CD를 키로 하는 해시맵 초기화
  const cdMap: Record<string, Region> = {};
 
  // 배열을 순회하며 각 항목을 해시맵에 추가
  regions.forEach(region => {
    // CD를 키로 사용하여 해시맵에 저장
    cdMap[region.cd] = region;
  });
 
  return cdMap;
}
